    add_integer("1", "2")

    print(e, end=", ")  # Expected Output: a must be an integer

try:
    add_integer(float('inf'), float('inf'))
except Exception as e:
    print(e, end=", ")  # Expected Output: float overflow

# Test case 5: Cannot convert float NaN to integer
try:
    add_integer(float('nan'), float('nan'))
except Exception as e:
    print(e, end=", ")  # Expected Output: cannot convert float NaN to integer
